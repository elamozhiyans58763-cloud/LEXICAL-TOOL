#include <stdio.h>
#include <ctype.h>
#include <string.h>

#define MAX 100

char stack[MAX];
int top = -1;

void push(char x) {
    stack[++top] = x;
}

char pop() {
    if (top == -1)
        return -1;
    return stack[top--];
}

int precedence(char x) {
    if (x == '+' || x == '-')
        return 1;
    if (x == '*' || x == '/')
        return 2;
    return 0;
}

void infixToPostfix(char *expr) {
    char output[MAX];
    int k = 0;

    for (int i = 0; expr[i] != '\0'; i++) {
        char ch = expr[i];

        if (isalnum(ch)) {
            output[k++] = ch;
        }
        else if (ch == '(') {
            push(ch);
        }
        else if (ch == ')') {
            while (top != -1 && stack[top] != '(') {
                output[k++] = pop();
            }
            pop();
        }
        else {
            while (top != -1 && precedence(stack[top]) >= precedence(ch)) {
                output[k++] = pop();
            }
            push(ch);
        }
    }

    while (top != -1) {
        output[k++] = pop();
    }

    output[k] = '\0';

    printf("Postfix: %s\n", output);
}

int main() {
    char expr[MAX];

    printf("Enter Infix Expression: ");
    scanf("%s", expr);

    infixToPostfix(expr);

    return 0;
}
